1. Two Sum

Given an array of integers, return indices of the two numbers such that they add up to a specific target.

You may assume that each input would have exactly one solution, and you may not use the same element twice.

Example:

Given nums = [2, 7, 11, 15], target = 9,

Because nums[0] + nums[1] = 2 + 7 = 9,
return [0, 1].

class Solution:
    def twoSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[int]
        """
        num_dict = {}
        for i in range(len(nums)):
            if num_dict.get(target-nums[i],None) == None:
                num_dict[nums[i]] = i
            else:
                return (num_dict[target - nums[i]], i)

2. Add Two Numbers

You are given two non-empty linked lists representing two non-negative integers. The digits are stored in reverse order and each of their nodes contain a single digit. Add the two numbers and return it as a linked list.

You may assume the two numbers do not contain any leading zero, except the number 0 itself.

Example

Input: (2 -> 4 -> 3) + (5 -> 6 -> 4)
Output: 7 -> 0 -> 8
Explanation: 342 + 465 = 807.

class Solution(object):
    def addTwoNumbers(self, l1, l2):
        """
        :type l1: ListNode
        :type l2: ListNode
        :rtype: ListNode
        """
        if l1 is None:
            return l2
        elif l2 is None:
            return l1
        else:          
            carry = 0
            ret =ListNode(0)
            ret_Last = ret
            
            while(l1 or l2):
                sum = 0
                if(l1):
                    sum = l1.val
                    l1 = l1.next
                if(l2):
                    sum += l2.val
                    l2 = l2.next
                sum += carry
                ret_Last.next = ListNode(sum%10)
                ret_Last = ret_Last.next
                carry = (sum >= 10)
            if(carry):
                ret_Last.next =ListNode(1)
            ret_Last = ret.next
            del ret
            return ret_Last

3. Longest Substring Without Repeating Characters

Given a string, find the length of the longest substring without repeating characters.

Examples:

Given "abcabcbb", the answer is "abc", which the length is 3.

Given "bbbbb", the answer is "b", with the length of 1.

Given "pwwkew", the answer is "wke", with the length of 3. Note that the answer must be a substring, "pwke" is a subsequence and not a substring.

class Solution:
    def lengthOfLongestSubstring(self, s):
        """
        :type s: str
        :rtype: int
        """
        know_dict = {}
        know_max = 0
        num = 0
        for i in s:
            if i not in know_dict:
                know_dict[i] = num
                num += 1
            else:
                r_num = 0
                for j in know_dict.keys():
                    if know_dict[j] <= know_dict[i]:
                        know_dict[j] = -1
                    else:
                        know_dict[j] = r_num
                        r_num += 1
                know_dict[i] = r_num
                num = r_num+1
            if num > know_max:
                know_max = num
        return know_max

LeetCode题解3：Longest Substring Without Repeating Characters

class Solution:
    def lengthOfLongestSubstring(self, s):
        max_length, start, char_dict = 0, 0, {}
        for idx, char in enumerate(s, 1):
            if char_dict.get(char, -1) >= start:
                start = char_dict[char]
            char_dict[char] = idx
            max_length = max(max_length, idx - start)
        return max_length

class Solution:
    def findErrorNums(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        n = len(nums)
        n_sum = (1 + n) * n // 2
        t = sum(set(nums))
        mis = n_sum - t
        dup = sum(nums) - t
        return [dup, mis]

Solve the Equation

Solve a given equation and return the value of x in the form of string “x=#value”. The equation contains only ‘+’, ‘-’ operation, the variable x and its coefficient.

If there is no solution for the equation, return “No solution”.

If there are infinite solutions for the equation, return “Infinite solutions”.

If there is exactly one solution for the equation, we ensure that the value of x is an integer.

class Solution:
    def solveEquation(self, equation):
        """
        :type equation: str
        :rtype: str
        """
        equation = equation.replace("+", " +")
        equation = equation.replace("-", " -")
        if "= -" in equation:
            equation = equation.replace("= -", "=-")
        else:
            equation = equation.replace("=", "=+")
        if equation[0] == " ":
            equation = equation[1:]
        else:
            equation = "+" + equation
        [left, right] = equation.split("=")
        x_num = 0
        left = left.split(" ")
        other_num = 0
        right = right.split(" ")
        for i in left:
            if "x" in i:
                if i == "+x":
                    x_num += 1
                elif i == "-x":
                    x_num -= 1
                else:
                    x_num += int(i[:-1])
            else:
                other_num += -int(i)
        for i in right:
            if 'x' in i:
                if i == "+x":
                    x_num -= 1
                elif i == "-x":
                    x_num += 1
                else:
                    x_num += -int(i[:-1])
            else:
                other_num += int(i)
        if x_num == 0 and other_num == 0:
            return "Infinite solutions"
        elif x_num == 0:
            return "No solution"
        else:
            return "x=" + str(other_num // x_num)

Minimum Moves to Equal Array Elements II

Given a non-empty integer array, find the minimum number of moves required to make all array elements equal, where a move is incrementing a selected element by 1 or decrementing a selected element by 1.

You may assume the array’s length is at most 10,000.

class Solution:
    def minMoves2(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        res = 0
        nums.sort()
        if len(nums)%2 == 0:
            for i in nums:
                res += abs(i-nums[len(nums)//2])
            return res
        else:
            res_1 = 0
            for i in nums:
                res += abs(i-nums[len(nums)//2-1])
                res_1 += abs(i-nums[len(nums)//2])
            return min(res,res_1)

Maximum Product of Three Numbers

Given an integer array, find three numbers whose product is maximum and output the maximum product.

class Solution:
    def maximumProduct(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if max(nums) > 0:
            result = max(nums)
            nums.remove(max(nums))
            nums.sort()
            return result * max(nums[0]*nums[1],nums[-1]*nums[-2]) 
        else:
            nums.sort()
            return nums[-1]*nums[-2]*nums[-3]

Count Numbers with Unique Digits

Given a non-negative integer n, count all numbers with unique digits, x, where 0 ≤ x < 10n.

Example: Given n = 2, return 91. (The answer should be the total numbers in the range of 0 ≤ x < 100, excluding [11,22,33,44,55,66,77,88,99])

class Solution:
    def countNumbersWithUniqueDigits(self, n):
        """
        :type n: int
        :rtype: int
        """
        if n == 0:
            return 1
        res = 0
        for i in range(n):
            if i == 0:
                res += 10
            else :
                res += self.count(i+1)
        return res
    
    def count(self,n):
        i = 9
        res = 9
        while i >= (9-n+2):
            res *= i
            i -= 1
        return res
