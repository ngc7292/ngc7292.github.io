}

vector<int> path[1001];
int cnt;
bool visit[1001];

void dfs(int i){
    if(i == cnt || visit[i]) return;
    visit[i] = true;
    for (int j : path[i]) {
        dfs(j);
    }
}
int main() {
    int n,m,k;
    scanf("%d %d %d",&n,&m,&k);
    int c1,c2;

        scanf("%d", &cnt);

        memset(visit, false, sizeof(visit));

                dfs(i);

            }
        }
